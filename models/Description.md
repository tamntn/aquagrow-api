## User ⏳
```javascript
{
    username: String,
    password: String,
    name: String,    
    picture: String,
    joined: Date,
    zipCode: String,
    zone: ObjectId.Zone,
    phone: String,
    system: ObjectId.System,
    portfolio: ObjectId.Portfolio,
    notifications: [ObjectId.Notification]
}
```
